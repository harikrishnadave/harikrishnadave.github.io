How It Works

Step 01 · Weekly Analysis

Market Bias Trend

Identify the macro trend using the Market Bias Ribbon — a dual EMA fill that turns green in bull trends and red in bear trends. Only trade in the direction of the bias.

Bull Trend — price above ribbon
Bear Trend — price below ribbon
Step 02 · Mode Detection

Market Mode

Dictate which "mode" the market is in before entering. Only enter trades during Compression or pullbacks — never chase Expansion moves.

Expansion — trending impulse
Capitulation — sell-off pressure
Compression — consolidation ✅ Enter
Step 03 · Confirmation

BX Trender Signal

Confirm bullish accumulation on the BX Trender oscillator. Look for Higher Highs (HH) or Higher Lows (HL) on the histogram before entering. This is the final gate.

HH / HL = Bullish Accumulation ✅
LH / LL = Bearish Re-Distribution ❌

Pine Script — TradingView Indicator

Two scripts to add to TradingView. Add the Market Bias Ribbon as an overlay on your price chart, and the BX Trender Oscillator in a separate pane below.

BX Trender Oscillator
// ═══════════════════════════════════════════════════════════
// Market Bias Ribbon
// Based on the Elite Trader Playbook
// Add this as an overlay on your price chart
// ═══════════════════════════════════════════════════════════
//@version=5
indicator("Market Bias Ribbon", shorttitle="MBR", overlay=true)

// ─── Inputs ────────────────────────────────────────────────
var GRP = "Ribbon Settings"
fast_len = input.int(8,  title="Fast EMA", minval=1, group=GRP)
slow_len = input.int(21, title="Slow EMA", minval=1, group=GRP)
show_signals = input.bool(true, "Show Trend Change Signals", group=GRP)

// ─── EMAs ───────────────────────────────────────────────────
ema_fast = ta.ema(close, fast_len)
ema_slow = ta.ema(close, slow_len)
is_bull  = ema_fast > ema_slow

// ─── Ribbon Fill ────────────────────────────────────────────
bull_col = color.new(#00c87a, 70)
bear_col = color.new(#ff4d6a, 70)

p1 = plot(ema_fast, "Fast EMA", color=color.new(color.gray, 100))
p2 = plot(ema_slow, "Slow EMA", color=color.new(color.gray, 100))
fill(p1, p2, color=is_bull ? bull_col : bear_col, title="Market Bias Ribbon")

// ─── EMA Lines ──────────────────────────────────────────────
plot(ema_fast, "Fast EMA",
     color=is_bull ? color.new(#00c87a, 0) : color.new(#ff4d6a, 0),
     linewidth=1)
plot(ema_slow, "Slow EMA",
     color=is_bull ? color.new(#00c87a, 30) : color.new(#ff4d6a, 30),
     linewidth=2)

// ─── Trend Change Signals ───────────────────────────────────
bull_cross = is_bull and not is_bull[1]
bear_cross = not is_bull and is_bull[1]

plotshape(show_signals and bull_cross,
          title="Bull Signal",
          location=location.belowbar,
          style=shape.triangleup,
          size=size.small,
          color=#00c87a)
plotshape(show_signals and bear_cross,
          title="Bear Signal",
          location=location.abovebar,
          style=shape.triangledown,
          size=size.small,
          color=#ff4d6a)
// ═══════════════════════════════════════════════════════════
// BX Trender Oscillator
// Based on the Elite Trader Playbook
// Add this as a NEW PANE below your price chart
// ═══════════════════════════════════════════════════════════
//@version=5
indicator("BX Trender Oscillator", shorttitle="BX-OSC", overlay=false)

// ─── Inputs ────────────────────────────────────────────────
var GRP1 = "BX Trender Settings"
short_l1 = input.int(5,  title="Short - L1", minval=1, group=GRP1)
short_l2 = input.int(20, title="Short - L2", minval=1, group=GRP1)
short_l3 = input.int(5,  title="Short - L3 (Smooth)", minval=1, group=GRP1)
long_l1  = input.int(20, title="Long  - L1", minval=1, group=GRP1)
long_l2  = input.int(5,  title="Long  - L2 (Smooth)", minval=1, group=GRP1)

var GRP2 = "Display"
show_labels = input.bool(true, "Show HH / HL / LH / LL Labels", group=GRP2)
show_aplus  = input.bool(true, "Show A+ Entry Signals", group=GRP2)
show_mode   = input.bool(true, "Show Market Mode Background", group=GRP2)

// ─── BX Trender Calculation ─────────────────────────────────
short_raw = ta.ema(close, short_l1) - ta.ema(close, short_l2)
short_pct = short_raw / close * 100
bx_short  = ta.ema(short_pct, short_l3)

long_raw  = ta.ema(close, long_l1) - ta.ema(close, short_l1)
long_pct  = long_raw / close * 100
bx_long   = ta.ema(long_pct, long_l2)

bx = bx_short + bx_long

// ─── HH / HL / LH / LL Detection ───────────────────────────
is_peak   = bx[1] > bx[0] and bx[1] > bx[2]
is_trough = bx[1] < bx[0] and bx[1] < bx[2]

var last_peak   = float(na)
var prev_peak   = float(na)
var last_trough = float(na)
var prev_trough = float(na)

if is_peak
    prev_peak  := last_peak
    last_peak  := bx[1]

if is_trough
    prev_trough := last_trough
    last_trough := bx[1]

is_HH = is_peak   and not na(prev_peak)   and bx[1] > prev_peak
is_LH = is_peak   and not na(prev_peak)   and bx[1] < prev_peak
is_HL = is_trough and not na(prev_trough) and bx[1] > prev_trough
is_LL = is_trough and not na(prev_trough) and bx[1] < prev_trough

// ─── Market Mode Detection ──────────────────────────────────
ema_fast   = ta.ema(close, 8)
ema_slow   = ta.ema(close, 21)
atr_val    = ta.atr(14)
atr_avg    = ta.sma(atr_val, 20)
range_high = ta.highest(high, 10)
range_low  = ta.lowest(low,  10)
range_pct  = (range_high - range_low) / range_low * 100

in_compression  = range_pct < 3.0 or atr_val < atr_avg * 0.8
in_capitulation = ema_fast < ema_slow and bx < -1.5
in_expansion    = math.abs(ema_fast - ema_slow) > math.abs(ema_fast[1] - ema_slow[1]) and atr_val > atr_avg

// ─── Colors ─────────────────────────────────────────────────
bar_col = bx >= 0
         ? (bx > bx[1] ? color.new(#00c87a, 0) : color.new(#00c87a, 45))
         : (bx < bx[1] ? color.new(#ff4d6a, 0) : color.new(#ff4d6a, 45))

// ─── Plot Histogram ─────────────────────────────────────────
plot(bx, "BX Trender", style=plot.style_histogram, color=bar_col, linewidth=2)
hline(0, "Zero Line", color=color.new(color.white, 70), linewidth=1)

// ─── Mode Background ────────────────────────────────────────
bgcolor(show_mode and in_compression  ? color.new(#4466ff, 90) : na, title="Compression")
bgcolor(show_mode and in_expansion    ? color.new(#00ff88, 94) : na, title="Expansion")
bgcolor(show_mode and in_capitulation ? color.new(#ff4444, 92) : na, title="Capitulation")

// ─── HH / HL / LH / LL Labels ───────────────────────────────
if show_labels
    if is_HH
        label.new(bar_index[1], bx[1], "HH ✅",
                  style=label.style_label_down,
                  color=color.new(#00c87a, 15),
                  textcolor=color.white, size=size.small)
    if is_HL
        label.new(bar_index[1], bx[1], "HL ✅",
                  style=label.style_label_up,
                  color=color.new(#00c87a, 15),
                  textcolor=color.white, size=size.small)
    if is_LH
        label.new(bar_index[1], bx[1], "LH 🔴",
                  style=label.style_label_down,
                  color=color.new(#ff4d6a, 15),
                  textcolor=color.white, size=size.small)
    if is_LL
        label.new(bar_index[1], bx[1], "LL 🔴",
                  style=label.style_label_up,
                  color=color.new(#ff4d6a, 15),
                  textcolor=color.white, size=size.small)

// ─── A+ Entry Signal ────────────────────────────────────────
// Fires when: prior compression + HL formed + BX crossing positive
a_plus_bull = show_aplus and in_compression[1] and is_HL and bx[1] > 0
if a_plus_bull
    label.new(bar_index[1], bx[1] - 0.5, "A+ 🎯",
              style=label.style_label_up,
              color=color.new(#ffd700, 0),
              textcolor=color.black, size=size.normal)
